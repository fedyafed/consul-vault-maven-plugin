5   package com.github.fedyafed;
6   
7   import com.ecwid.consul.v1.ConsulClient;
8   import com.ecwid.consul.v1.kv.model.GetValue;
9   import org.apache.maven.plugin.AbstractMojo;
10  import org.apache.maven.plugin.MojoExecutionException;
11  import org.apache.maven.plugin.logging.Log;
12  import org.apache.maven.plugins.annotations.LifecyclePhase;
13  import org.apache.maven.plugins.annotations.Mojo;
14  import org.apache.maven.plugins.annotations.Parameter;
15  import org.apache.maven.project.MavenProject;
16  
17  import java.util.Collections;
18  import java.util.List;
19  import java.util.Properties;
20  
21  import static java.nio.charset.StandardCharsets.UTF_8;
22  
23  /**
24   * Goal read a value from Consul KV by given key.
25   */
26  @Mojo(name = "read-consul", defaultPhase = LifecyclePhase.INITIALIZE)
27  public class ReadConsulMojo extends AbstractMojo {
28  
29      @Parameter(defaultValue = "${project}", required = true, readonly = true)
30      private MavenProject project;
31  
32      /**
33       * Consul KV key prefixes.
34       */
35      @Parameter(defaultValue = "${consul.prefixes}")
36      private List<String> prefixes = Collections.singletonList("config");
37  
38      /**
39       * Consul hostname.
40       */
41      @Parameter(defaultValue = "${consul.host}")
42      private String host = "localhost";
43  
44      /**
45       * Consul HTTP port.
46       */
47      @Parameter(defaultValue = "${consul.port}")
48      private int port = 8500;
49      private ConsulClient client;
50  
51      /**
52       * Mojo execution.
53       *
54       * @throws MojoExecutionException - default Mojo exception.
55       */
56      public void execute() throws MojoExecutionException {
57          client = new ConsulClient(host, port);
58  
59          for (String prefix : prefixes) {
60              setProperties(prefix);
61          }
62      }
63  
64      private void setProperties(String prefix) {
65          Log log = getLog();
66          if (!prefix.endsWith("/")) {
67              prefix += '/';
68          }
69  
70          List<GetValue> values = client.getKVValues(prefix).getValue();
71          if (values != null) {
72              Properties properties = project.getProperties();
73              for (GetValue value : values) {
74                  String key = getPropertyKey(value.getKey(), prefix);
75                  String decodedValue = value.getDecodedValue(UTF_8);
76                  log.debug("Key:\t" + key + ", value:\t" + decodedValue);
77                  properties.setProperty(key, decodedValue);
78              }
79              log.info("Found properties:\t" + values.size() + " for prefix:\t" + prefix);
80          } else {
81              log.warn("Key prefix\t'" + prefix + "' not found.");
82          }
83      }
84  
85      static String getPropertyKey(String key, String prefix) {
86          return key
87                  .substring(prefix.length())
88                  .replace('/', '.');
89      }
90  
91      public List<String> getPrefixes() {
92          return prefixes;
93      }
94  
95      public void setPrefixes(List<String> prefixes) {
96          this.prefixes = prefixes;
97      }
98  
99      String getHost() {
100         return host;
101     }
102 
103     void setHost(String host) {
104         this.host = host;
105     }
106 
107     int getPort() {
108         return port;
109     }
110 
111     void setPort(int port) {
112         this.port = port;
113     }
114 
115     void setProject(MavenProject project) {
116         this.project = project;
117     }
118 }
